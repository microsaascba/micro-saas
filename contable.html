function calcularCierreMes() { const periodo = document.getElementById('cierreMesInput').value; if(!periodo) return; // SIMULACIÓN DE PERFIL: Esto debería venir de la configuración del usuario // true = Monotributista | false = Responsable Inscripto const esMonotributista = true; const ventasMes = dbSales.filter(s => (s.date || '').startsWith(periodo)); const gastosMes = dbExpenses.filter(e => (e.date || '').startsWith(periodo)); let ventasBrutas = 0; let cmvTotal = 0; let ivaDebito = 0; let gastosBrutos = 0; let ivaCreditoTotal = 0; // --- PROCESO DE VENTAS --- ventasMes.forEach(v => { ventasBrutas += v.total; const items = typeof v.items === 'string' ? JSON.parse(v.items) : (v.items || []); items.forEach(i => { cmvTotal += (i.qty * (i.cost || 0)); }); // Si no es monotributista, calculamos IVA sobre venta (aquí podrías usar la alícuota por producto si la guardas) if (!esMonotributista) { ivaDebito += (v.total - (v.total / 1.21)); } }); // --- PROCESO DE GASTOS (Punto Crítico) --- gastosMes.forEach(g => { gastosBrutos += g.amount; if (!esMonotributista) { // Usamos el dato real tipeado, no el calculado por división ivaCreditoTotal += (g.ivaAmount || 0); } }); // --- LÓGICA DE MOSTRAR RESULTADOS --- const ventasNetasReal = esMonotributista ? ventasBrutas : (ventasBrutas - ivaDebito); const gastosNetosReal = esMonotributista ? gastosBrutos : (gastosBrutos - ivaCreditoTotal); const utilidadBruta = ventasNetasReal - cmvTotal; const utilidadNeta = utilidadBruta - gastosNetosReal; // Actualización de UI (Labels dinámicos) document.getElementById('cmVentasBrutas').textContent = currency(ventasBrutas); document.getElementById('cmIvaDebito').textContent = esMonotributista ? "N/A (Monotributo)" : currency(ivaDebito); document.getElementById('cmVentasNetas').textContent = currency(ventasNetasReal); document.getElementById('cmGastosBrutos').textContent = currency(gastosBrutos); document.getElementById('cmIvaCredito').textContent = esMonotributista ? "N/A (Costo total)" : currency(ivaCreditoTotal); document.getElementById('cmGastosNetos').textContent = currency(gastosNetosReal); // Resultado final const elUtilidadNeta = document.getElementById('cmUtilidadNeta'); elUtilidadNeta.textContent = currency(utilidadNeta); elUtilidadNeta.style.color = utilidadNeta < 0 ? '#fca5a5' : '#86efac'; // Advertencia Fiscal (Indispensable para blindarte legalmente) const posIvaBox = document.getElementById('cmPosicionIvaBox'); if (esMonotributista) { posIvaBox.innerHTML = `
Perfil Monotributista: Los impuestos se pagan por categoría fija. Este reporte es puramente gerencial.
`; } else { const posicionIva = ivaDebito - ivaCreditoTotal; document.getElementById('cmPosicionIvaMonto').textContent = currency(Math.abs(posicionIva)); document.getElementById('cmPosicionIvaTexto').textContent = posicionIva > 0 ? "A Pagar IVA (Estimado):" : "Saldo a Favor IVA:"; } } // Función auxiliar para el modal function sumarTotalGasto() { const n = parseFloat(document.getElementById('expNeto').value) || 0; const i = parseFloat(document.getElementById('expIva').value) || 0; const o = parseFloat(document.getElementById('expOtros').value) || 0; document.getElementById('expAmount').value = (n + i + o).toFixed(2); }
